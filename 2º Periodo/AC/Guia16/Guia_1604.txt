JMP MAIN

FIND_MIN:
MOV A, M
MOV D, A
DCR C
INX H
LOOP_MIN:
MOV A, C
ORA A
JZ END_MIN
MOV A, M
CMP D
JNC NEXT
MOV D, A
NEXT:
INX H
DCR C
JMP LOOP_MIN
END_MIN:
MOV A, D
RET

MAIN:
LXI H, 0050H
MVI C, 05H
CALL FIND_MIN
STA 0055H
HLT
